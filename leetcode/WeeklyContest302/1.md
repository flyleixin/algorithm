```c++
class Solution {
public:
    vector<int> numberOfPairs(vector<int>& nums) {
        int vis[105] = {0};
        for(int i = 0;i<nums.size();i++){
            vis[nums[i]] ++;
        }
        int ans = 0;
        for(int i = 0;i<=100;i++){
            ans+=(vis[i]/2);
        }
        vector<int>ve;
        ve.push_back(ans);
        ve.push_back(nums.size()-2*ans);
        return ve;
    }
};
```
